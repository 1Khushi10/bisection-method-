#include<iostream.h>
#include<conio.h>
#include<math.h>
#define f(x) x*x*x-x-1
#define e 0.0001
void main()
{
float x0,x1,x,f0,f1,f2 ;
int i=1;
clrscr();
do
{
cout<<"enter initial roots:";
cin>>x0>>x1;
}while(f(x0)*f(x1)>0);
do
{
f0=f(x0);
f1=f(x1);
x=((x0+x1)/2);
f2=f(x);
if(f2<0)
{
x0=x;
f0=f2;
}
else
{
x1=x;
f1=f2;
}
cout<<"iteration:" <<i++<<"\t";
cout<<"root ="<<x<<"\t";
cout<<"value of root"<< f2<<endl;
}while(fabs(f2)>e);
cout<<"final root is:"<<x;
getch();
}
